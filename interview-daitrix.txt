print("Hello")

arr = [1, 2, 4, 6, 3, 7, 8]
N = len(arr)
zeroArray = [ 0 for x in arr]
# print(zeroArray)

for i in range(N-1):
    # print(arr[i])
    zeroArray[arr[i]-1] = 1
    # print(zeroArray)
    
ans = 0
for ind, z in enumerate(zeroArray):
    if z == 0:
        ans = ind+1
print(ans)
